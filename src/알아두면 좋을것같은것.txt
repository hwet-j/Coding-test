-- 짝수로 이루어진 데이터를 출력

--> 초기 풀이
-- emp 테이블에서 사원번호가 짝수인 직원만 출력
SELECT *
FROM emp
WHERE empno = ROUND(empno / 2) * 2;

--> 인터넷에서 나온 나머지 구하는 함수 MOD 사용
SELECT *
FROM emp
WHERE MOD(empno,2) = 0;


-- 홀수로 이루어진 데이터를 출력

--> 초기 풀이
-- emp 테이블에서 사원번호가 짝수인 직원만 출력
SELECT *
FROM emp
WHERE empno = ROUND(empno / 2) * 2 - 1;

--> 인터넷에서 나온 나머지 구하는 함수 MOD 사용
SELECT *
FROM emp
WHERE MOD(empno,2) = 1;



-- 제곱과 제곱근을 구하는 방법

-- 풀이 못함
-- 문제를 못찾아 비슷한 문제로 에시
-- empno에서 deptno의 제곱을 뺸 값에 루트씌운 값을 소수점 4자리까지만 출력하라
SELECT ROUND(SQRT(empno-POWER(deptno,2)), 4)
FROM emp;
